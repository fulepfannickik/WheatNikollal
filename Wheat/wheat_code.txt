using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Windows.Forms;

namespace Wheat
{
    public partial class Form1 : Form
    {
        List<Country> countries = new List<Country>();
        List<string> years = new List<string>();

        public Form1()
        {
            InitializeComponent();
        }

        private void megnyitásToolStripMenuItem_Click(object sender, EventArgs e)
        {
            OpenFileDialog openFileDialog = new OpenFileDialog();
            if (openFileDialog.ShowDialog() != DialogResult.OK) return;

            string path = openFileDialog.FileName;

            countries.Clear();
            years.Clear();

            using (StreamReader reader = new StreamReader(path))
            {
                string headerLine = reader.ReadLine();
                years = headerLine.Split(';').Skip(1).ToList();

                while (!reader.EndOfStream)
                {
                    string line = reader.ReadLine();
                    string[] parts = line.Split(';');

                    Country country = new Country
                    {
                        Name = parts[0]
                    };

                    for (int i = 1; i < parts.Length; i++)
                    {
                        string year = years[i - 1];
                        string value = parts[i];

                        if (value == ":" || string.IsNullOrWhiteSpace(value))
                        {
                            country.Data[year] = null;
                        }
                        else
                        {
                            country.Data[year] = double.Parse(value);
                        }
                    }

                    countries.Add(country);
                }
            }

            Showdata();
        }

        private void Showdata()
        {
            WheatDataGrid.Rows.Clear();
            WheatDataGrid.Columns.Clear();

            if (years.Count == 0) return;

            WheatDataGrid.Columns.Add("Country", "Country");

            foreach (var year in years)
            {
                WheatDataGrid.Columns.Add(year, year);
            }

            foreach (var country in countries)
            {
                List<object> row = new List<object>();
                row.Add(country.Name);

                foreach (var year in years)
                {
                    row.Add(country.Data.ContainsKey(year) ? country.Data[year] : null);
                }

                WheatDataGrid.Rows.Add(row.ToArray());
            }

            WheatDataGrid.AllowUserToAddRows = false;
            WheatDataGrid.AutoResizeColumns();
        }
    }

    internal class Country
    {
        public string Name { get; set; }
        public Dictionary<string, double?> Data { get; set; } = new Dictionary<string, double?>();
    }
}
